            mins = mins - 1
        end
        secs = secs - 1
        wait(1)

    textLabel.Text = "Ready to submit"
end)

_G.AutoMessages = false
local minWait = 2 
local maxWait = 10

local function autoSpeak(message)
    local ohString1 = message
    local ohString2 = "All"
    game:GetService("ReplicatedStorage").DefaultChatSystemChatEvents.SayMessageRequest:FireServer(ohString1, ohString2)
end

local MessageSettings = {}

    table.insert(MessageSettings, t)
end)

messagesTab:Textbox("Wait Time (Seconds)", true, function(t)
    minWait = tonumber(t) or minWait
    maxWait = minWait * 2
end)

MiscTab:Button("Auto Claim Season Rewards", function(t)
    while t do
        for i = 1, 9 do
            local args = {
                [1] = i
            }
            game:GetService("ReplicatedStorage"):WaitForChild("Remotes"):WaitForChild("Season"):FireServer(unpack(args))
            wait(1)
        end
        wait(300)
    end
end)

MiscTab:Toggle("Anti AFK", false, function(toggle)
    local virtualUser = game:GetService("VirtualUser")
    local player = game:GetService("Players").LocalPlayer
    local antiAfk = false
    if toggle and not antiAfk then
        antiAfk = true
        player.Idled:Connect(function()
            virtualUser:CaptureController()
            virtualUser:ClickButton2(Vector2.new())
        end)
    else
        antiAfk = false
        player.Idled:Disconnect()
    end
end)
